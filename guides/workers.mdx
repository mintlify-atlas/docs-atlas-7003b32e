## Overview

SMyEG uses dedicated background workers to process asynchronous tasks like shapefile imports, geometry recalculations, land use variations, and measurement campaign calculations. Workers run as separate Node.js processes and poll the database for pending jobs.

## Worker Architecture

### Available Workers

### Worker Pattern

Both workers follow a polling pattern:

1. Query database for pending jobs
2. Lock job by updating status to `PROCESSING`
3. Execute job logic
4. Update job with result (`COMPLETED` or `FAILED`)
5. Sleep for configured interval
6. Repeat

## Geo Worker Scheduler

### Purpose

The geo worker processes three types of jobs:

- **Import Jobs** - Extract and validate shapefiles, import geometries
- **Recalc Jobs** - Recalculate area and centroids for existing geometries  
- **Variation Jobs** - Create land use variation records in bulk

### Worker Execution Loop

From `src/workers/geo-worker-scheduler.ts:18-64`:

```typescript
async function runBatch() {
  if (running || stopped) return;
  running = true;

  try {
    let importProcessed = 0;
    let recalcProcessed = 0;
    let variationProcessed = 0;

    // Process import jobs
    for (let i = 0; i < importBatchSize; i += 1) {
      const result = await processNextPendingImportJob();
      if (!result.processed) break;
      importProcessed += 1;
    }

    // Process recalc jobs
    for (let i = 0; i < recalcBatchSize; i += 1) {
      const result = await processNextRecalcJob();
      if (!result.processed) break;
      recalcProcessed += 1;
    }

    // Process variation jobs
    for (let i = 0; i < variationBatchSize; i += 1) {
      const result = await processNextGeoVariationJob();
      if (!result.processed) break;
      variationProcessed += 1;
    }

    if (importProcessed > 0 || recalcProcessed > 0 || variationProcessed > 0) {
      console.info(
        `[${now()}] [geo-worker] processed import_jobs=${importProcessed} ` +
        `recalc_jobs=${recalcProcessed} variation_jobs=${variationProcessed}`
      );
    }
  } catch (error) {
    console.error(`[${now()}] [geo-worker] batch error`, error);
  } finally {
    running = false;
  }
}
```

```typescript
export async function processNextRecalcJob() {
  const job = await prisma.forestGeometryRecalcJob.findFirst({
    where: {
      status: GeoRecalcJobStatus.PENDING,
      runAfter: { lte: new Date() },
    },
    orderBy: { createdAt: 'asc' },
  });

  if (!job) {
    return { processed: false, reason: 'no_pending_recalc_jobs' };
  }

  // Mark as processing
  await prisma.forestGeometryRecalcJob.update({
    where: { id: job.id },
    data: {
      status: GeoRecalcJobStatus.PROCESSING,
      startedAt: new Date(),
      attempts: { increment: 1 },
    },
  });

  // Query PostGIS for recalculated metrics
  const metrics = await prisma.$queryRaw`
    SELECT
      superficie_ha::double precision AS superficie_ha,
      ST_Y(centroid) AS lat,
      ST_X(centroid) AS lon
    FROM public.forest_geometry_n4
    WHERE organization_id = ${job.organizationId}::uuid
      AND level4_id = ${job.level4Id}::uuid
      AND is_active = TRUE
    ORDER BY valid_from DESC
    LIMIT 1
  `;

  // Update Level 4 with new metrics
  await tx.forestPatrimonyLevel4.update({
    where: { id: job.level4Id },
    data: {
      totalAreaHa: metric.superficie_ha,
      centroidLatitude: metric.lat,
      centroidLongitude: metric.lon,
    },
  });

  // Sync land use totals
  await syncLandUseSurfaceTotals(tx, job.organizationId);
}
```

### Variation Job Processing

From `src/lib/geo-import-worker.ts:687-806`, variation jobs create bulk land use change records:

```typescript
export async function processNextGeoVariationJob() {
  const job = await prisma.geoLandVariationJob.findFirst({
    where: {
      status: GeoVariationJobStatus.PENDING,
      runAfter: { lte: new Date() },
    },
    orderBy: { createdAt: 'asc' },
  });

  if (!job) return { processed: false };

  const payload = job.payload as { items?: GeoVariationPayloadItem[] };
  const items = Array.isArray(payload?.items) ? payload.items : [];

  for (const item of items) {
    const level4 = await prisma.forestPatrimonyLevel4.findFirst({
      where: {
        id: level4Id,
        level3: { level2: { organizationId: job.organizationId } },
      },
    });

    if (!level4) continue;

    await prisma.landPatrimonialVariation.create({
      data: {
        organizationId: job.organizationId,
        level4Id: level4.id,
        previousLandUseName,
        newLandUseName,
        affectedAreaHa,
        variationDate: job.variationDate,
        notes: job.notes,
        kind: 'INCREMENTO',
        status: 'PENDIENTE',
        createdById: job.createdById,
      },
    });
  }
}
```

```typescript
async function shutdown(signal: string) {
  if (stopped) return;
  stopped = true;
  console.info(`[${now()}] [geo-worker] stopping scheduler (${signal})...`);

  try {
    await prisma.$disconnect();
  } finally {
    process.exit(0);
  }
}

process.on('SIGINT', () => {
  clearInterval(timer);
  void shutdown('SIGINT');
});

process.on('SIGTERM', () => {
  clearInterval(timer);
  void shutdown('SIGTERM');
});
```

### Protocol Calculation Example: BIO_SOBREVIVENCIA

From `src/lib/asset-measurement-worker.ts:154-306`:

```typescript
function buildBioSurvivalResult(observations: WorkerObservation[]) {
  let sumLiveTotal = 0;
  let sumDeadTotal = 0;
  let sumMissingCount = 0;
  let sumExpectedPlants = 0;
  const liveTotalsPerParcel: number[] = [];

  // Validate and accumulate each observation
  for (const observation of observations) {
    const payload = asObject(observation.payload);
    const liveNormal = readNumber(payload, 'liveNormal');
    const liveAtipic = readNumber(payload, 'liveAtipic') ?? 0;
    // ... read all fields

    const liveTotal = liveNormal + liveAtipic + liveDrought + livePest;
    const deadTotal = deadPest + deadDrought + deadMech;
    const evaluatedPoints = liveTotal + deadTotal + missingCount;
    const expectedPoints = 100 / (spacingL1 * spacingL2);

    // Validation
    if (Math.abs(evaluatedPoints - expectedPoints) > 1) {
      invalidObservationCount += 1;
      continue;
    }

    sumLiveTotal += liveTotal;
    sumDeadTotal += deadTotal;
    liveTotalsPerParcel.push(liveTotal);
  }

  // Calculate metrics
  const totalObserved = sumLiveTotal + sumDeadTotal + sumMissingCount;
  const survivalRealPct = (sumLiveTotal / totalObserved) * 100;
  const densityVivaHa = (sumLiveTotal / validObservationCount) * 100;

  // Calculate sampling error using t-student
  const mean = liveTotalsPerParcel.reduce((a, b) => a + b, 0) / n;
  const variance = liveTotalsPerParcel.reduce((a, v) => a + (v - mean) ** 2, 0) / (n - 1);
  const stdDev = Math.sqrt(variance);
  const tValue = tStudent95ByDegrees(n - 1);
  const samplingErrorPct = ((tValue * (stdDev / Math.sqrt(n))) / mean) * 100;

  // Decision logic
  const survivalSemaforo = classifySurvivalSemaforo(survivalRealPct);
  const survivalDecision = classifySurvivalDecision(survivalRealPct);

  return {
    protocolType: MeasurementProtocolType.BIO_SOBREVIVENCIA,
    metrics: {
      SURVIVAL_REAL_PCT: roundOrNull(survivalRealPct, 2),
      DENSITY_VIVA_HA: roundOrNull(densityVivaHa, 0),
      SAMPLING_ERROR_PCT: roundOrNull(samplingErrorPct, 2),
      SURVIVAL_SEMAFORO: survivalSemaforo,
      OPERATIONAL_ACTION: survivalDecision.operationalAction,
    },
    quality: {
      validObservationCount,
      invalidObservationCount,
      zeroLiveObservationCount,
    },
  };
}
```

### Job Statuses

| Status | Description |
|--------|-------------|
| `PENDING` | Queued, waiting for worker to pick up |
| `PROCESSING` | Currently being processed by worker |
| `COMPLETED` | Successfully completed |
| `FAILED` | Processing failed (see `errorMessage`) |

## Troubleshooting

### Worker Not Processing Jobs

<Steps>
  <Step title="Check Worker is Running">
    Verify process is running: `pm2 status` or `docker ps`
  </Step>
  <Step title="Check Database Connection">
    Ensure `DATABASE_URL` is correct and database is accessible
  </Step>
  <Step title="Check Job Status">
    Query jobs table to ensure jobs are in `PENDING` status, not stuck in `PROCESSING`
  </Step>
  <Step title="Review Logs">
    Check worker stdout/stderr for errors: `pm2 logs` or `docker logs`
  </Step>
</Steps>

### Jobs Stuck in PROCESSING

If a worker crashes mid-job, jobs may be stuck in `PROCESSING` status. Manually reset:

```sql
UPDATE "GeoImportJob"
SET status = 'PENDING',
    "startedAt" = NULL
WHERE status = 'PROCESSING'
  AND "startedAt" < NOW() - INTERVAL '10 minutes';
```

### High Memory Usage

Reduce batch sizes to process fewer jobs per cycle:

```bash
export GEO_IMPORT_BATCH_SIZE=2
export ASSET_MEASUREMENT_WORKER_BATCH_SIZE=10
```
